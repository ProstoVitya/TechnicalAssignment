SELECT P1.* FROM Period P1
OUTER APPLY
(SELECT TOP 1 * FROM Period P2 WHERE P1.PeriodID > P2.PeriodID ORDER BY P2.PeriodID DESC) P2
WHERE P1.Value<>P2.Value OR P2.Value IS NULL

--2
DELETE P1 FROM Period P1
OUTER APPLY
(SELECT TOP 1 * FROM Period P2 WHERE P1.PeriodID > P2.PeriodID ORDER BY P2.PeriodID DESC) P2
WHERE P1.Value=P2.Value
